#include <stdio.h>
#include <math.h>

int main(void) {
  float x1,y1,x2,y2,x3,y3;
  float l1,l2,l3;
  float tmp;
  printf("Digite o 1 ponto:");
  scanf("%f",&x1);
  printf("Digite o 1 ponto:");
  scanf("%f",&y1);

  printf("Digite o 2 ponto:");
  scanf("%f",&x2);
  printf("Digite o 2 ponto:");
  scanf("%f",&y2);

  printf("Digite o 3 ponto:");
  scanf("%f",&x3);
  printf("Digite o 3 ponto:");
  scanf("%f",&y3);

  l1=sqrt(pow(x1-x2,2)+pow(y1-y2,2));
  l2=sqrt(pow(x1-x3,2)+pow(y1-y3,2));
  l3=sqrt(pow(x3-x2,2)+pow(y3-y2,2));

  if(l1<l2){
    tmp = l1;l1=l2;l2=tmp;
  }
  if(l1<l3){
    tmp = l1;l1=l3;l3=tmp;
  }
  if(l1<l2+l3){
    printf("Os pontos formam um triangulo");
  }
  else{
    printf("Os pontos nao formam um triangulo");
  }
  if((l1==l2)&&(l2==l3)){
    printf(" equilatero\n");
  }
  else if((l1==l2)||(l2==l3)||(l1==l3)){
    printf(" isoceles\n");
  }
  else{
    printf(" escaleno\n");
  }



  return 0;
}
